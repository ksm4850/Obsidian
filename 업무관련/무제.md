#!/bin/bash

VERSION=$1

if [ -z "$VERSION" ]; then
    echo "사용법 deploy.sh <version>"
else
    GITLAB_REGISTRY="git.pilab.co.kr:5050/slam-1.0.0"

    cd /home/pi/slam_workspace/slam-be-fastapi

    echo "프론트 이미지 확인 중: ${GITLAB_REGISTRY}/slam-fe-nextjs:${VERSION}"
    docker manifest inspect ${GITLAB_REGISTRY}/slam-fe-nextjs:${VERSION} > /dev/null 2>&1

    if [ $? -eq 0 ]; then
        echo "프론트 이미지 배포 진행"
        docker pull ${GITLAB_REGISTRY}/slam-fe-nextjs:${VERSION}
        docker tag ${GITLAB_REGISTRY}/slam-fe-nextjs:${VERSION} slam_fe:${VERSION}
        docker tag ${GITLAB_REGISTRY}/slam-fe-nextjs:${VERSION} slam_fe
        docker compose down slam_fe
        docker compose up slam_fe -d
        echo "프론트 배포 완료"
    else
        echo "프론트 이미지를 찾을 수 없습니다: ${VERSION}"
    fi

    echo "백엔드 이미지 확인 중: ${GITLAB_REGISTRY}/slam-be-fastapi:${VERSION}"
    docker manifest inspect ${GITLAB_REGISTRY}/slam-be-fastapi:${VERSION} > /dev/null 2>&1
    if [ $? -eq 0 ]; then
        echo "백엔드 이미지 배포 진행"
        docker pull ${GITLAB_REGISTRY}/slam-be-fastapi:${VERSION}
        docker tag ${GITLAB_REGISTRY}/slam-be-fastapi:${VERSION} slam_be:${VERSION}
        docker tag ${GITLAB_REGISTRY}/slam-be-fastapi:${VERSION} slam_be 
        docker compose down slam_be 

        echo "백엔드 패치 스크립트 실행"
        docker compose run --rm slam_be_patch
	    docker compose up slam_be -d
        echo "백엔드 배포 완료"
    else
        echo "백엔드 이미지를 찾을 수 없습니다: ${VERSION}"
    fi
fi
